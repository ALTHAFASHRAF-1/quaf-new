Instructions

Google Sheet Format

Your Google Sheet must have these columns in order:

date | heading | description | videoLink | order

YouTube URL Conversion

The system automatically converts:

  • https://youtu.be/ID → https://www.youtube.com/embed/ID
  • https://youtube.com/watch?v=ID → https://www.youtube.com/embed/ID
  • https://www.youtube.com/embed/ID → (stays the same)

Get Script Code

Copy the Google Apps Script code from the button below:

Google Apps Script Code

// Google Apps Script for QUAF Video Showcase
// Paste this code in Google Sheets Script Editor

// Configuration
const SHEET_NAME = "Videos";
const HEADERS = ["date", "heading", "description", "videoLink", "order"];

function doGet(e) {
  return handleRequest(e);
}

function doPost(e) {
  return handleRequest(e);
}

function handleRequest(e) {
  const action = e.parameter.action;
  
  try {
    if (action === "addVideo") {
      return addVideo(e);
    } else if (action === "getVideos") {
      return getVideos();
    } else if (action === "deleteVideo") {
      return deleteVideo(e);
    } else if (action === "updateVideo") {
      return updateVideo(e);
    } else {
      return createResponse("error", "Invalid action", null);
    }
  } catch (error) {
    return createResponse("error", error.toString(), null);
  }
}

function addVideo(e) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sheet = ss.getSheetByName(SHEET_NAME) || ss.insertSheet(SHEET_NAME);
  
  // Set headers if sheet is empty
  if (sheet.getLastRow() === 0) {
    sheet.appendRow(HEADERS);
  }
  
  // Get data from request
  const data = {
    date: e.parameter.date,
    heading: e.parameter.heading,
    description: e.parameter.description,
    videoLink: convertToEmbed(e.parameter.youtubeUrl),
    order: parseInt(e.parameter.order)
  };
  
  // Validate data
  if (!data.date || !data.heading || !data.videoLink || !data.order) {
    return createResponse("error", "Missing required fields", null);
  }
  
  // Add to sheet
  const rowData = [data.date, data.heading, data.description, data.videoLink, data.order];
  sheet.appendRow(rowData);
  
  // Get the new row number
  const newRow = sheet.getLastRow();
  
  return createResponse("success", "Video added successfully", {
    id: newRow,
    ...data
  });
}

function getVideos() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sheet = ss.getSheetByName(SHEET_NAME);
  
  if (!sheet || sheet.getLastRow() === 0) {
    return createResponse("success", "No videos found", []);
  }
  
  const data = sheet.getDataRange().getValues();
  const headers = data[0];
  const videos = [];
  
  // Start from row 1 to skip header
  for (let i = 1; i < data.length; i++) {
    const row = data[i];
    const video = {
      id: i + 1, // Google Sheets rows are 1-indexed
      date: row[0] || "",
      heading: row[1] || "",
      description: row[2] || "",
      videoLink: row[3] || "",
      order: row[4] || 999
    };
    
    // Only add if it has a title
    if (video.heading) {
      videos.push(video);
    }
  }
  
  // Sort by order
  videos.sort((a, b) => a.order - b.order);
  
  return createResponse("success", "Videos retrieved", videos);
}

function deleteVideo(e) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sheet = ss.getSheetByName(SHEET_NAME);
  
  if (!sheet) {
    return createResponse("error", "Sheet not found", null);
  }
  
  const rowId = parseInt(e.parameter.id);
  
  if (rowId <= 1) {
    return createResponse("error", "Cannot delete header row", null);
  }
  
  sheet.deleteRow(rowId);
  
  return createResponse("success", "Video deleted successfully", { id: rowId });
}

function updateVideo(e) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sheet = ss.getSheetByName(SHEET_NAME);
  
  if (!sheet) {
    return createResponse("error", "Sheet not found", null);
  }
  
  const rowId = parseInt(e.parameter.id);
  
  if (rowId <= 1) {
    return createResponse("error", "Cannot update header row", null);
  }
  
  const data = {
    date: e.parameter.date,
    heading: e.parameter.heading,
    description: e.parameter.description,
    videoLink: convertToEmbed(e.parameter.youtubeUrl),
    order: parseInt(e.parameter.order)
  };
  
  // Update row
  const rowRange = sheet.getRange(rowId, 1, 1, 5);
  rowRange.setValues([[data.date, data.heading, data.description, data.videoLink, data.order]]);
  
  return createResponse("success", "Video updated successfully", {
    id: rowId,
    ...data
  });
}

function convertToEmbed(youtubeUrl) {
  if (!youtubeUrl) return "";
  
  // Handle various YouTube URL formats
  let videoId = "";
  
  // Standard formats
  if (youtubeUrl.includes("youtu.be/")) {
    videoId = youtubeUrl.split("youtu.be/")[1];
  } else if (youtubeUrl.includes("youtube.com/watch?v=")) {
    videoId = youtubeUrl.split("v=")[1];
  } else if (youtubeUrl.includes("youtube.com/embed/")) {
    videoId = youtubeUrl.split("embed/")[1];
  }
  
  // Clean up any extra parameters
  if (videoId.includes("&")) {
    videoId = videoId.split("&")[0];
  }
  if (videoId.includes("?")) {
    videoId = videoId.split("?")[0];
  }
  
  return videoId ? `https://www.youtube.com/embed/${videoId}` : youtubeUrl;
}

function createResponse(status, message, data) {
  const output = ContentService.createTextOutput();
  output.setContent(JSON.stringify({
    status: status,
    message: message,
    data: data
  }));
  output.setMimeType(ContentService.MimeType.JSON);
  return output;
}

Deployment Instructions

  1. Copy the entire code above
  2. Open your Google Sheet
  3. Go to Extensions → Apps Script
  4. Delete any existing code and paste this code
  5. Click "Save" (disk icon or Ctrl+S)
  6. Click "Deploy" → "New deployment"
  7. Select "Web app" as type
  8. Set "Execute as" to "Me"
  9. Set "Who has access" to "Anyone"
  10. Click "Deploy" and copy the web app URL
  11. Paste the URL in the Settings tab
